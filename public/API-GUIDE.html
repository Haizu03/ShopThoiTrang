🗄️ Kết nối SQL Server

File: config/database.js

const sql = require('mssql/msnodesqlv8');

const config = {
    connectionString: 'Driver={ODBC Driver 18 for SQL Server};Server=(localdb)\\MSSQLLocalDB;Database=shop_thoi_trang;Trusted_Connection=yes;TrustServerCertificate=yes;'
};

let pool;

async function getPool() {
    if (!pool) {
        pool = await new sql.ConnectionPool(config).connect();
        console.log('Ket noi SQL Server thanh cong!');
    }
    return pool;
}

module.exports = { sql, getPool };

Tham số kết nối

Tham số Giá trị Mô tả
Server (localdb)\MSSQLLocalDB Local database (máy dev). Thay bằng IP/hostname cho production
Database shop_thoi_trang Tên database
Trusted_Connection yes Sử dụng Windows Authentication (không cần username/password)
TrustServerCertificate yes Bỏ qua SSL certificate validation
💡 Ghi chú: ConnectionPool tự động cache kết nối. Chỉ tạo 1 pool duy nhất trong ứng dụng.

🔐 API Xác thực (routes/auth.js)

POST /dang-nhap

Mô tả: Đăng nhập tài khoản

Body:

{ "ten_dang_nhap": "Lan123", "mat_khau": "Lan123" }

Response: Redirect đến `/dashboard` (admin) hoặc `/trang-chu` (user)

Session: Lưu req.session.user = {id, ten_dang_nhap, ho_ten, vai_tro}

POST /dang-ky

Mô tả: Đăng ký tài khoản mới (vai_tro = 'nguoidung')

Body:

{ "ho_ten": "Nguyễn Văn A", "ten_dang_nhap": "nguyena", "mat_khau": "123456", "xac_nhan": "123456" }

Validation: Mật khẩu ≥ 6 ký tự, khớp xác nhận, tên đăng nhập chưa tồn tại

Password Hash: bcryptjs salt=10

GET /dang-xuat

Mô tả: Đăng xuất

Action: Xoá session, redirect `/dang-nhap`

Code Query (routes/auth.js): 
const result = await pool.request()

🛒 API Đơn hàng (routes/donHang.js, routes/user.js)

Admin quản lý đơn hàng

GET /don-hang

Mô tả: Danh sách đơn hàng (admin only)

Render: views/don-hang/index.ejs

User đặt hàng

POST /dat-hang

Mô tả: Tạo đơn hàng mới (user đã đăng nhập)

Body:

{
  "ho_ten": "Nguyễn Văn A",
  "so_dien_thoai": "0901234567",
  "email": "a@gmail.com",
  "dia_chi": "123 Nguyễn Huệ, Q.1, TP.HCM",
  "ghi_chu": "Ghi chú...",
  "phuong_thuc_tt": "tien_mat",
  "ma_giam_gia_id": 1,
  "cart_items": [
    { "san_pham_id": 1, "so_luong": 2 }
  ]
}

Logic:

  • Validate mã giảm giá (nếu có)
  • Tạo khách hàng nếu chưa tồn tại
  • Tạo đơn hàng + chi tiết (transaction)
  • Cập nhật tồn kho sản phẩm
  • Xoá giỏ hàng session
SQL Transactions (routes/user.js): 
const transaction = new sql.Transaction(pool);
await transaction.begin();

// Tạo đơn hàng
const insertDH = await new sql.Request(transaction)
    .input('ma_dh', sql.NVarChar, maDH)
    .input('khach_hang_id', sql.Int, khachHangId)
    .input('tai_khoan_id', sql.Int, req.session.user.id)
    .query(`INSERT INTO don_hang (ma_dh, khach_hang_id, tai_khoan_id, tong_tien, giam_gia, thanh_toan, phuong_thuc_tt)
        OUTPUT INSERTED.id VALUES (...)`);

// Thêm chi tiết
for (const item of chiTietItems) {
    await new sql.Request(transaction)
        .query('INSERT INTO chi_tiet_don_hang (...) VALUES (...)');
    
    // Cập nhật tồn kho
    await new sql.Request(transaction)
        .query('UPDATE san_pham SET so_luong = so_luong - @so_luong WHERE id = @id');
}

await transaction.commit();

Lỗi SQL phổ biến

Lỗi Nguyên nhân Sửa
ConnectionError: connect ECONNREFUSED Không kết nối được SQL Server Kiểm tra server đang chạy: sqllocaldb start MSSQLLocalDB
Unique constraint violation Dữ liệu bị trùng lặp Check tên đăng nhập, mã sản phẩm, mã đơn hàng
Foreign key constraint violation ID tham chiếu không tồn tại Kiểm tra danh_muc_id, khach_hang_id, san_pham_id có tồn tại
Timeout or connection lost Truy vấn quá chậm hoặc server ngắt kết nối Thêm index, optimize query, tăng timeout
⚠️ Cảnh báo: Tất cả các route đều sử dụng parameterized queries để tránh SQL injection: 
pool.request()
    .input('ten_dang_nhap', sql.NVarChar, userInput)
    .query('SELECT * FROM tai_khoan WHERE ten_dang_nhap = @ten_dang_nhap')
Đừng bao giờ dùng string concatenation!
✅ Best Practices:
  • Sử dụng parameterized queries cho mọi SQL statement
  • Validate input dữ liệu trước khi insert/update
  • Xử lý lỗi try-catch trong mọi async function
  • Log lỗi chi tiết cho debugging
  • Sử dụng transactions cho các hoạt động multi-step
